import { IBusinessWithProcessors } from './../../../shared/to/IBusinessWithProcessors';
import { IProcessorDbUpdate } from './../../../shared/to/IProcessorUpdate';
import { BusinessService } from './../../services/business.service';
import {
  BUSINESS_LOADED_ACTION, BUSINESS_UPDATE_IMG_ACTION, BUSINESS_UPLOAD_FILE_ACTION, BusinessFileUploadedAction,
  BusinessImgUpdatedAction,
  BusinessLoadedAction,
  ErrorOcurredAction,
  LOAD_BUSINESS_ACTION,
  ProcessorUpdatedAction,
  UPDATE_PROCESSOR_ACTION,
} from '../actions';
import { IApplicationState } from '../app-state';
import { IUiState } from '../ui-state';

import { Store } from '@ngrx/store';
import {Actions, Effect } from '@ngrx/effects';
import { Injectable } from '@angular/core';
import { Action } from '@ngrx/store';
import { Observable } from 'rxjs/Observable';
import { ToastrService } from "ngx-toastr";





@Injectable()
export class BusinessEffectService{

    constructor(
        private action$: Actions,
        private businessService: BusinessService,
        private store: Store<IApplicationState>,
        private toastrService: ToastrService ){}

    /**
     *
     */
    @Effect() loadBusiness$: Observable<Action> = this.action$
        .ofType(LOAD_BUSINESS_ACTION)
        .debug("Loading business from the server")
        .withLatestFrom(this.store.select("uiState"))
        .map(([any, uiState])=> uiState)
        .filter((uiState: any) => uiState.currentUser)
        .switchMap((uiState: any) => this.businessService.getBusinessById(uiState.currentUser.business))
        .debug("Business loaded from the server")
        .map(business => new BusinessLoadedAction(business))
        .catch(err => Observable.of(new ErrorOcurredAction(`Error al cargar informacin del negocio: ${err}`)));

    /**
     *
     */
    @Effect() updateProcessors$: Observable<Action> = this.action$
        .ofType(UPDATE_PROCESSOR_ACTION)
        .debug("Updating processor i9 (mamalon) in the server")
        .withLatestFrom(this.store.select("uiState"))
        .map(([action, uiState])=> {
            return {
                'uiState': uiState,
                'action': action
            };
        })
        .filter((payload: {uiState: any, action: Action}) => payload.uiState.currentUser)
        .switchMap(payload => this.businessService.updateBusinessProcessors(payload.uiState.currentUser.business, payload.action.payload))
        .debug("Processors succesfully updated in the db dao repository")
        .do( update => {
            this.toastrService.success(`Procesador actualizado`, "Exito");
            return update;
        })
        .map((processors: IProcessorDbUpdate) => new ProcessorUpdatedAction(processors))
        .catch(err => Observable.of(new ErrorOcurredAction(`Error al actualizar el procesador: ${err}`)));

    @Effect() changeBusinessImage$ : Observable<Action> = this.action$
      .ofType(BUSINESS_UPDATE_IMG_ACTION)
      .debug("Uploading new business image")
      .withLatestFrom(this.store.select("storeData")).map(([action, storeData])=> {
        return {
          'storeData': storeData,
          'action': action
        };
      })
      .filter((payload: {storeData: any, action: Action}) => payload.storeData.business._id)
      .switchMap((payload)=>Observable.zip(...[Observable.of(payload), this.businessService.getImgSignedRequest(payload.action.payload.file, payload.action.payload.businessId)]))
      .debug("Signed request obtained")
      .switchMap(data => Observable.zip(...[Observable.of(data[0]), Observable.of(data[1]), this.businessService.uploadImgFile(data[0].action.payload.file, data[1].signedRequest, data[1].url)]))
      .debug("Image uploaded to AWS")
      .do((data)=>this.toastrService.success("Imagen actualizada satisfactoriamente"))
      .map((data)=> new BusinessImgUpdatedAction(({businessId: data[0].storeData.business._id, url: data[1].url})))
      .catch(err => Observable.of(new ErrorOcurredAction(`Error al generar signed request: ${err}`)));

  @Effect() uploadBusinessFile : Observable<Action> = this.action$
    .ofType(BUSINESS_UPLOAD_FILE_ACTION)
    .debug("Uploading new business file")
    .withLatestFrom(this.store.select("storeData")).map(([action, storeData])=> {
      return {
        'storeData': storeData,
        'action': action
      };
    })
    .filter((payload: {storeData: any, action: Action}) => payload.storeData.business)
    .switchMap((payload)=>Observable.zip(...[Observable.of(payload), this.businessService.getFileSignedRequest(payload.action.payload.file, payload.action.payload.businessId)]))
    .debug("Signed request obtained")
    .switchMap(data => Observable.zip(...[Observable.of(data[0]), Observable.of(data[1]), this.businessService.uploadFile(data[0].action.payload.file, data[1].returnData.signedRequest)]))
    .debug("File uploaded to AWS")
    .do((data)=>this.toastrService.success("Archivo subido satisfactoriamente STORE"))
    .map((data)=>  new BusinessFileUploadedAction(({businessId: data[0].storeData.business._id, files: data[1].files})))
    .catch(err => Observable.of(new ErrorOcurredAction(`Error al generar signed request: ${err}`)));

}
