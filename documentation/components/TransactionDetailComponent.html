import { UpdateTransactionAction, LoadTransactionAction } from '../../../../store/actions';
import {
    UpdateTransactionModalComponent,
} from '../../components/update-transaction-modal/update-transaction-modal.component';
import { Transaction, TransactionStatus } from '../../../../../shared/models/transaction';
import { IApplicationState } from '../../../../store/app-state';
import { ITransactionSummary } from '../../../../../shared/vm/ITransactionSummary';
import { Component, OnDestroy, OnInit, ViewChild } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { Store } from '@ngrx/store';
import { mapStateToTransactionSummary } from '../../../../store/mappers/mapStateToTransactionSummaries';
import { Observable, Subscription } from 'rxjs/Rx';
import { User } from "../../../../../shared/models/user";
import { TransactionService } from "../../../../services/transaction.service";

@Component({
  selector: 'omni-transaction-detail',
  templateUrl: './transaction-detail.component.html',
  styleUrls: ['./transaction-detail.component.scss']
})
export class TransactionDetailComponent implements OnInit, OnDestroy {
  private sub: Subscription;
  transaction: Observable<ITransactionSummary>;
  currentUser: Observable<User>;
  status: number;
  additionalInfo : any;
  @ViewChild('updateTransactionModal') updateTransactionModal: UpdateTransactionModalComponent;


  constructor(private route: ActivatedRoute, private store: Store<IApplicationState>,
              private tS : TransactionService) {
  }

  ngOnInit() {
    this.sub = this.route.params.subscribe(params => {
      let processor = params['processor'];
      let id = params['id'];
      //if(processor == 0)
      setTimeout(()=>this.store.dispatch(new LoadTransactionAction(id)), 0);
      this.transaction = this.store.select(state => mapStateToTransactionSummary(state, id, processor));
    });
    this.currentUser = this.store.select(state => state.uiState.currentUser);
  }

  ngOnDestroy(){
    this.sub.unsubscribe();
  }

  onCancelTransaction(){
    this.status = TransactionStatus.Cancelado;
    this.updateTransactionModal.showTransactionModal();
  }

  onClarifyTransaction(){
    this.status = TransactionStatus.Aclarando;
    this.updateTransactionModal.showTransactionModal();
  }

  onUpdateTransaction(){
    this.status = -1;
    this.updateTransactionModal.showTransactionModal();
  }

  onNewTransactionUpdate(event){
    setTimeout(()=> {this.store.dispatch(new UpdateTransactionAction(event))}, 0);
  }

  onModalCanceled(){
    console.log('update transaction canceled');
  }
}
