import { Subscription } from 'rxjs/Subscription';
import { Transaction, TransactionStatus } from '../../../../shared/models/transaction';
import { Store } from '@ngrx/store';
import { IApplicationState } from '../../../store/app-state';
import { Component, OnInit } from '@angular/core';

import { Router } from '@angular/router';
import { Observable } from "rxjs/Observable";
import * as moment from 'moment';
@Component({
  templateUrl: './statistics.component.html',
  styleUrls: ['./statistics.component.scss']
})
export class StatisticsComponent implements OnInit {
  transactions: Array<Transaction>;
  oneWeekAgo = moment().subtract(7, 'days');
  sub: Subscription;
  constructor(private store: Store<IApplicationState> ) {
    let completedTransactions = [];
    let otherTransactions = [];
    this.sub = this.store.select(state => state.storeData.business).subscribe(
      business => {
        if(business){
          business.transactions.filter( transaction => moment(transaction.createdAt).isAfter(this.oneWeekAgo)).forEach( transaction => {
            if(transaction.statusLog[transaction.statusLog.length -1 ].statusType === TransactionStatus.Procesado){
              completedTransactions.push(transaction);
            } else {
              otherTransactions.push(transaction);
            }
          });
          console.log("completadas", completedTransactions);
          console.log("otras", otherTransactions);
        }
      }
    ); 
   }

  }

  // convert Hex to RGBA
  public convertHex(hex: string, opacity: number) {
    hex = hex.replace('#', '');
    const r = parseInt(hex.substring(0, 2), 16);
    const g = parseInt(hex.substring(2, 4), 16);
    const b = parseInt(hex.substring(4, 6), 16);

    const rgba = 'rgba(' + r + ', ' + g + ', ' + b + ', ' + opacity / 100 + ')';
    return rgba;
  }
