import { TourModalComponent } from '../../../shared/components/tour-modal/tour-modal.component';
import { fadeIn, fadeOut, fadeUp } from '../../../../animations/animations';
import {
    TermsAndConditionsModalComponent,
} from '../../../shared/components/terms-and-conditions-modal/terms-and-conditions-modal.component';
import { IProcessorOption } from '../../../../../shared/vm/IProcessorSummary';
import { mapStateToAvaiableProcessors } from '../../../../store/mappers/mapStateToAvailableProcessors';
import { Business } from '../../../../../shared/models/business';
import { IApplicationState } from './../../../../store/app-state';

import {Component, OnInit, ViewChild} from '@angular/core';
import { Store } from '@ngrx/store';
import { Observable } from 'rxjs/Observable';
import { FileUploader } from "ng2-file-upload";
import { ToastrService } from "ngx-toastr";
import { Http } from "@angular/http";
import {BusinessUpdateImgAction, BusinessUploadFileAction} from "../../../../store/actions";
import {BusinessService} from "../../../../services/business.service";
import { ViewImageModalComponent } from "../../../../modules/shared/components/view-image-modal/view-image-modal.component";



@Component({
  selector: 'omni-business-config',
  templateUrl: './business-config.component.html',
  styleUrls: ['./business-config.component.scss'],
  animations: [
    fadeIn,
    fadeUp,
    fadeOut
  ]
})
export class BusinessConfigComponent implements OnInit {
  @ViewChild('pop') pop;
  @ViewChild('pop2') pop2;
  @ViewChild('pop3') pop3;
  @ViewChild('pop4') pop4;
  @ViewChild('pop5') pop5;
  @ViewChild('pop6') pop6;
  @ViewChild('fileView') fileView;
  @ViewChild('tourModal') tourModal: TourModalComponent;
  @ViewChild('viewImageModal') viewImageModal: ViewImageModalComponent;
  @ViewChild('termsAndConditions') termsAndConditionsModal: TermsAndConditionsModalComponent;  
  business$: Observable<Business>;
  availableProcessors$: Observable<IProcessorOption[]>;
  editName: boolean;
  editImage: boolean;
  editFile: boolean;
  fileSelected : string = 'Ningn archivo seleccionado';
  imgFileSelected : string = 'Ningn archivo seleccionado';
  businessImgUrl : string;
  businessId : string;
  uploadedFiles : Array<string>;
  URL : string = '';
  public uploader : FileUploader;
  public imgUploader : FileUploader;
  constructor(private store: Store<IApplicationState>, private toasterService : ToastrService, private http : Http, private businessService : BusinessService) {
    this.editName = false;
    this.editImage = false;
    this.editFile = false;
    this.business$ = this.store.select(state => state.storeData.business);
    this.availableProcessors$ = this.store.select(state=> mapStateToAvaiableProcessors(state));
   }

  ngOnInit() {
    this.business$.subscribe((business)=> {
      if(business){
        this.uploadedFiles = business.uploadedFiles;
        this.businessId = business["_id"];
        this.businessImgUrl = business.imgUrl;
        this.URL = `/api/user/upload/${this.businessId}`;
        this.uploader = new FileUploader({url: this.URL, itemAlias: 'file'});
        this.imgUploader = new FileUploader({url: this.URL, itemAlias: 'file'});
        this.uploader.onAfterAddingFile = (file)=> { this.fileSelected = file.file.name; file.withCredentials = false; };
        this.imgUploader.onAfterAddingFile = (file)=> { this.imgFileSelected = file.file.name; file.withCredentials = false; };

        this.uploader.onCompleteItem = (item:any, response:any, status:any, headers:any) => {
          console.log("ImageUpload:uploaded:", item, status, response);
          this.toasterService.success("Archivo subido satisfactoriamente");
        };
      }
    });

  }

  onChangeName(f, valid){
    if(valid){
      console.log(f.businessName);
      this.editImage = false;
    }
  }

  onChangeImage(){

  }

  onViewTermsAndConditions(e){
    this.termsAndConditionsModal.showTermsAndConditionsModal();
  }

  onViewTour(e){
    this.tourModal.showTourModal();
  }

  onViewImage(e){
    this.viewImageModal.showImageModal();
  }

  onViewImageCanceled(e){
    console.log("View image canceled");
  }

  mamazonUpload(type : number, fileName : any){
    if(type === 0){
      const files = document.getElementById('file-input')["files"];
      const file = files[0];
      if(file === undefined) {
        this.toasterService.warning('Favor de seleccionar un archivo', 'Ningn archivo seleccionado');
        return;
      }
      this.store.dispatch(new BusinessUploadFileAction({file, businessId : this.businessId}));
      this.editFile = false;
    }

    else{
      const files = document.getElementById('img-input')["files"];
      const file = files[0];
      if(file === undefined) {
        this.toasterService.warning('Favor de seleccionar un archivo', 'Ningn archivo seleccionado');
        return;
      }
      this.store.dispatch(new BusinessUpdateImgAction({file, businessId : this.businessId}));
      this.imgFileSelected = 'Ningn archivo seleccionado';
      this.editImage = false;

    }

  }


  getFile(file : string): void{
    this.businessService.getFile(file, this.businessId).subscribe(url=>window.open(url), e=> console.log(e));
  }

  
  onStartTour(step: number): void {
    switch(step){
      case 1:
        this.pop.show();
        this.pop2.hide();
        break;
      case 2:
        this.pop2.show();
        this.pop.hide();
        this.pop3.hide();
        break;
      case 3:
        this.pop3.show();
        this.pop2.hide();
        this.pop4.hide();
        break;
      case 4:
        this.pop4.show();
        this.pop3.hide();
        this.pop5.hide();
        break;
      case 5:
        this.pop5.show();
        this.pop4.hide();
        this.pop6.hide();
        break;
      case 6:
        this.pop6.show();
        this.pop5.hide();
        break;
    }
  }

}
