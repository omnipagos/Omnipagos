import { fadeIn, fadeUp, slideOut } from '../../animations/animations';
import { ViewImageModalComponent } from '../../modules/shared/components/view-image-modal/view-image-modal.component';
import { mapStateToPaymentsByProcessorTotal } from '../../store/mappers/mapStateToPaymentsByProcessor';
import { IPaymentByProcessor } from './../../../shared/vm/IPaymentByProcessor';
import { LoadBusinessAction } from './../../store/actions';
import { IApplicationState } from './../../store/app-state';
import { mapStateToTransactionsByStatusTotals } from './../../store/mappers/mapStateToTransactionsByStatus';
import { mapStateToTransactionSummaries } from './../../store/mappers/mapStateToTransactionSummaries';
import { ITransactionsByStatus } from './../../../shared/vm/ITransactionByStatus';
import { ITransactionSummary } from './../../../shared/vm/ITransactionSummary';
import { Component, HostBinding, OnInit, ViewChild } from '@angular/core';
import { Store } from '@ngrx/store';
import * as moment from 'moment';
import { Observable, Subscription } from 'rxjs/Rx';


@Component({
  selector: 'omni-dashboard',
  templateUrl: './dashboard.component.html',
  animations: [
    fadeIn,
    fadeUp,
    slideOut
  ],
  styleUrls: ['./dashboard.component.scss']
})
export class DashboardComponent implements OnInit {
  @HostBinding('@slideOut') routerAnimation = true;
  @ViewChild('pop') pop;
  @ViewChild('pop2') pop2;
  @ViewChild('pop3') pop3;
  @ViewChild('pop4') pop4;
  @ViewChild('pop5') pop5;
  @ViewChild('pop6') pop6;
  @ViewChild('viewImageModal') viewImageModal: ViewImageModalComponent;
  businessName: Observable<string>;
  businessImgUrl: Observable<string>;
  transactions: Observable<Array<ITransactionSummary>>;
  groupedTransactions: Observable<Array<ITransactionsByStatus>>;
  groupedPayments: Observable<Array<any>>;
  isAdmin: Observable<boolean>;
  today: Date = new Date();

  constructor(private store: Store<IApplicationState>) {
  }

  ngOnInit() {
    setTimeout(() => this.store
      .takeLast(1)
      .subscribe(state => {
        if (!state.storeData.business) this.store.dispatch(new LoadBusinessAction())
    }), 1000);
    this.businessName = this.store.select(state => state.storeData.business ? state.storeData.business.name : '');
    this.transactions = this.store.select(state => mapStateToTransactionSummaries(state));
    this.groupedTransactions = this.store.select(state => mapStateToTransactionsByStatusTotals(state));
    this.groupedPayments = this.store.select(state => mapStateToPaymentsByProcessorTotal(state));
    this.businessImgUrl = this.store.select(state => state.storeData.business ? state.storeData.business.imgUrl : '');
    this.isAdmin = this.store.select(state => state.uiState.currentUser ? state.uiState.currentUser.isAdmin: false );
  }

  days: Array<moment.Moment>;
  public brandPrimary = '#2980b9';
  public brandSuccess = '#27ae60';
  public brandInfo = '#3498db';
  public brandWarning = '#f1c40f';
  public brandDanger = '#c0392b';
  // dropdown buttons
  public status: { isopen } = { isopen: false };
  public toggleDropdown($event: MouseEvent): void {
    $event.preventDefault();
    $event.stopPropagation();
    this.status.isopen = !this.status.isopen;
  }

  getDays(): Array<moment.Moment> {
    const temp = [];
    for (let i = 0; i < 7; i++) {
      temp.push(moment().subtract(i, 'day'));
    }
    console.log(temp);
    return temp;
  }

  onViewBusinessImage(e): void{
    this.viewImageModal.showImageModal();
  }

  onStartTour(step: number): void {
    switch(step){
      case 1:
        this.pop.show();
        this.pop2.hide();
        break;
      case 2:
        this.pop2.show();
        this.pop.hide();
        this.pop3.hide();
        break;
      case 3:
        this.pop3.show();
        this.pop2.hide();
        this.pop4.hide();
        break;
      case 4:
        this.pop4.show();
        this.pop3.hide();
        this.pop5.hide();
        break;
      case 5:
        this.pop5.show();
        this.pop4.hide();
        this.pop6.hide();
        break;
      case 6:
        this.pop6.show();
        this.pop5.hide();
        break;
    }

  }
  

  getWidgetClass(processorNo: number): string {
    switch(processorNo){
      case  1 :
        return 'col-lg-12';
      case  2 :
        return 'col-lg-6';
      case 3 :
        return 'col-lg-6 col-xl-4';
      case 4 :
        return 'col-lg-6 col-xl-3';
    }
  }

  onViewImageCanceled(e):void {
    console.log("View image closed");
  }




}
